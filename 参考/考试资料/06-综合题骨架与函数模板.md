这一章不讲概念，直接按“考试题怎么落地”来写。

- 调一个电影 API
- 把电影保存到 SQLite
- 做登录
- 用户可以收藏电影
- 页面用 Thymeleaf 展示

### 21.8 API Service

```java
@Service
public class RemoteMovieService {

    private final RestTemplate restTemplate;

    @Value("${movie.api.base-url}")
    private String movieApiBaseUrl;

    @Value("${movie.api-key}")
    private String movieApiKey;

    public RemoteMovieService(RestTemplate restTemplate) {
        this.restTemplate = restTemplate;
    }

    public RemoteMoviePageDTO getPopularMovies(int pageNumber) {
        String url = movieApiBaseUrl
                + "/movie/popular?page=" + pageNumber
                + "&api_key=" + movieApiKey
                + "&language=fr-FR";
        return restTemplate.getForObject(url, RemoteMoviePageDTO.class);
    }

    public RemoteMovieDTO getMovieById(Long movieId) {
        String url = movieApiBaseUrl
                + "/movie/" + movieId
                + "?api_key=" + movieApiKey
                + "&language=fr-FR";
        return restTemplate.getForObject(url, RemoteMovieDTO.class);
    }
}
```

### 21.9 数据库存储 Service

```java
@Service
public class StoredMovieService {

    private final StoredMovieRepository storedMovieRepository;
    private final RemoteMovieService remoteMovieService;

    public StoredMovieService(
            StoredMovieRepository storedMovieRepository,
            RemoteMovieService remoteMovieService) {
        this.storedMovieRepository = storedMovieRepository;
        this.remoteMovieService = remoteMovieService;
    }

    public List<StoredMovieEntity> findAll() {
        return storedMovieRepository.findAll();
    }

    public StoredMovieEntity findById(Long movieId) {
        return storedMovieRepository.findById(movieId).orElse(null);
    }

    public StoredMovieEntity saveMovieFromApi(Long movieId) {
        StoredMovieEntity existingMovie = findById(movieId);
        if (existingMovie != null) {
            return existingMovie;
        }

        RemoteMovieDTO remoteMovieDTO = remoteMovieService.getMovieById(movieId);
        StoredMovieEntity storedMovie = convertToEntity(remoteMovieDTO);
        return storedMovieRepository.save(storedMovie);
    }

    private StoredMovieEntity convertToEntity(RemoteMovieDTO remoteMovieDTO) {
        StoredMovieEntity storedMovie = new StoredMovieEntity();
        storedMovie.setId(remoteMovieDTO.getId());
        storedMovie.setTitle(remoteMovieDTO.getTitle());
        storedMovie.setReleaseDate(remoteMovieDTO.getRelease_date());
        storedMovie.setVoteAverage(remoteMovieDTO.getVote_average());
        storedMovie.setOverview(remoteMovieDTO.getOverview());
        return storedMovie;
    }
}
```

### 21.10 用户 Service

```java
@Service
public class AppMemberService {

    private final AppMemberRepository appMemberRepository;
    private final PasswordEncoder passwordEncoder;

    public AppMemberService(
            AppMemberRepository appMemberRepository,
            PasswordEncoder passwordEncoder) {
        this.appMemberRepository = appMemberRepository;
        this.passwordEncoder = passwordEncoder;
    }

    public AppMemberEntity findByEmail(String email) {
        return appMemberRepository.findByEmail(email);
    }

    public AppMemberEntity registerMember(String fullName, String email, String rawPassword) {
        if (fullName == null || fullName.isBlank()) {
            throw new IllegalArgumentException("Nom obligatoire");
        }
        if (email == null || email.isBlank()) {
            throw new IllegalArgumentException("Email obligatoire");
        }
        if (rawPassword == null || rawPassword.isBlank()) {
            throw new IllegalArgumentException("Mot de passe obligatoire");
        }
        if (appMemberRepository.existsByEmail(email)) {
            throw new IllegalArgumentException("Email deja utilise");
        }

        AppMemberEntity appMember = new AppMemberEntity();
        appMember.setFullName(fullName);
        appMember.setEmail(email);
        appMember.setPasswordHash(passwordEncoder.encode(rawPassword));
        return appMemberRepository.save(appMember);
    }
}
```

### 21.11 收藏 Service

```java
@Service
public class MemberMoviePreferenceService {

    private final MemberMoviePreferenceRepository memberMoviePreferenceRepository;
    private final AppMemberRepository appMemberRepository;
    private final StoredMovieService storedMovieService;

    public MemberMoviePreferenceService(
            MemberMoviePreferenceRepository memberMoviePreferenceRepository,
            AppMemberRepository appMemberRepository,
            StoredMovieService storedMovieService) {
        this.memberMoviePreferenceRepository = memberMoviePreferenceRepository;
        this.appMemberRepository = appMemberRepository;
        this.storedMovieService = storedMovieService;
    }

    public void addFavorite(String email, Long movieId) {
        AppMemberEntity appMember = appMemberRepository.findByEmail(email);
        StoredMovieEntity storedMovie = storedMovieService.saveMovieFromApi(movieId);

        MemberMoviePreferenceEntity preference =
                memberMoviePreferenceRepository.findByAppMember_IdAndStoredMovie_Id(
                        appMember.getId(), storedMovie.getId());

        if (preference == null) {
            preference = new MemberMoviePreferenceEntity();
            preference.setAppMember(appMember);
            preference.setStoredMovie(storedMovie);
            preference.setFavorite(false);
            preference.setWatchlist(false);
        }

        preference.setFavorite(true);
        memberMoviePreferenceRepository.save(preference);
    }
}
```

```java
public class AuthController {

    private final AppMemberService appMemberService;

    public AuthController(AppMemberService appMemberService) {

    }

    @GetMapping("/login")
    public String loginPage() {
        return "login";

        return "register";
    }

    @PostMapping("/register")
    public String register(
            @RequestParam("fullName") String fullName,
            @RequestParam("email") String email,
            @RequestParam("password") String password,

        try {
            appMemberService.registerMember(fullName, email, password);

            redirectAttributes.addFlashAttribute("errorMessage", exception.getMessage());
            return "redirect:/register";
        }
    }
}
```

### 21.13 页面 Controller

```java
@Controller
@RequestMapping("/movies")
public class StoredMovieController {

    private final RemoteMovieService remoteMovieService;
    private final StoredMovieService storedMovieService;
    private final MemberMoviePreferenceService memberMoviePreferenceService;

    public StoredMovieController(
            RemoteMovieService remoteMovieService,
            StoredMovieService storedMovieService,
            MemberMoviePreferenceService memberMoviePreferenceService) {
        this.remoteMovieService = remoteMovieService;
        this.storedMovieService = storedMovieService;
        this.memberMoviePreferenceService = memberMoviePreferenceService;
    }

    @GetMapping
    public String movieList(@RequestParam(defaultValue = "1") Integer page, Model model) {
        RemoteMoviePageDTO remoteMoviePage = remoteMovieService.getPopularMovies(page);
        model.addAttribute("moviePage", remoteMoviePage);
        model.addAttribute("movies", remoteMoviePage.getResults());
        return "movie-list";
    }

    @GetMapping("/{movieId}")
    public String movieDetail(@PathVariable Long movieId, Model model) {
        model.addAttribute("movie", remoteMovieService.getMovieById(movieId));
        return "movie-detail";
    }

    @PostMapping("/{movieId}/favorite")
    public String addFavorite(@PathVariable Long movieId, Principal principal) {
        memberMoviePreferenceService.addFavorite(principal.getName(), movieId);
        return "redirect:/movies/" + movieId;
    }

    @GetMapping("/stored")
    public String storedMovies(Model model) {
        model.addAttribute("storedMovies", storedMovieService.findAll());
        return "stored-movie-list";
    }
}
```

### 21.15 Security 配置

```java
@Configuration
public class SecurityConfig {

    @Bean
    public UserDetailsService userDetailsService(AppMemberRepository appMemberRepository) {
        return username -> {
            AppMemberEntity appMember = appMemberRepository.findByEmail(username);
            if (appMember == null) {
                throw new UsernameNotFoundException("Utilisateur introuvable");
            }

            return org.springframework.security.core.userdetails.User
                    .withUsername(appMember.getEmail())
                    .password(appMember.getPasswordHash())
                    .roles("USER")
                    .build();
        };
    }

    @Bean
    public SecurityFilterChain securityFilterChain(HttpSecurity http) throws Exception {
        http
            .authorizeHttpRequests(auth -> auth
                .requestMatchers("/login", "/register", "/error").permitAll()
                .anyRequest().authenticated()
            )
            .formLogin(form -> form
                .loginPage("/login")
                .usernameParameter("email")
                .defaultSuccessUrl("/movies", true)
                .failureUrl("/login?error")
            )
            .logout(logout -> logout.logoutSuccessUrl("/login?logout"));

        return http.build();
    }
}
```
